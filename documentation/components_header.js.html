/**
 * Baut den Header (Logo, Slogan, Hilfe-Link, Profil-Icon + Menücontainer).
 * @function renderHeader
 * @param {string} [initials="GU"] Initialen für den Profilkreis
 * @returns {string} HTML
 */
export function renderHeader(initials = "GU") {
  return /*html*/ `
    <header>
      <h2 class="header-slogan">Kanban Project Management Tool</h2>
      <img class="header-logo" src="./img/img/header_logo.svg" alt="Join Logo" />
      <div class="header-icons">
        <div class="help-icon">
          <a href="#help" aria-label="Hilfe öffnen">
            <img src="./img/img/help_icon.svg" alt="Help" />
          </a>
        </div>
        <div class="profile-icon" id="profileIcon" aria-haspopup="true" aria-expanded="false" tabindex="0">
          <div class="profile-name ${
            initials === "GU" ? "guest-font-size" : ""
          }" id="profileInitials">${escapeHtml(initials)}</div>
        </div>
      </div>
    </header>
    <div class="sub-menu-bg d-none" id="profileMenuBackdrop"></div>
    <div class="sub-menu sub-menu-close" id="profileMenu" aria-hidden="true">
      <a href="#profile"><div>Profil</div></a>
      <a href="#settings"><div>Einstellungen</div></a>
      <a href="#logout" id="logoutLink"><div>Logout</div></a>
    </div>
  `;
}

/**
 * Escaped Sonderzeichen für sichere Ausgabe im DOM.
 * @private
 * @function escapeHtml
 * @param {string} str Eingabetext
 * @returns {string} Escaped Text
 */
function escapeHtml(str) {
  return str.replace(
    /[&<>"]+/g,
    (c) => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;" }[c])
  );
}

/**
 * Initialisiert Auf-/Zuklappen des Profil-Menüs inkl. Tastatursteuerung.
 * @function initProfileMenu
 * @returns {void}
 */
export function initProfileMenu() {
  const icon = document.getElementById("profileIcon");
  const menu = document.getElementById("profileMenu");
  const backdrop = document.getElementById("profileMenuBackdrop");
  if (!icon || !menu || !backdrop) return;

  function open() {
    menu.classList.remove("sub-menu-close");
    menu.classList.add("sub-menu-open");
    backdrop.classList.remove("d-none");
    icon.setAttribute("aria-expanded", "true");
    document.body.classList.add("sub-menu-bg-open");
  }

  function close() {
    menu.classList.add("sub-menu-close");
    menu.classList.remove("sub-menu-open");
    backdrop.classList.add("d-none");
    icon.setAttribute("aria-expanded", "false");
    document.body.classList.remove("sub-menu-bg-open");
  }

  icon.addEventListener("click", () =>
    menu.classList.contains("sub-menu-open") ? close() : open()
  );
  icon.addEventListener("keydown", (e) => {
    if (e.key === "Enter" || e.key === " ") {
      e.preventDefault();
      menu.classList.contains("sub-menu-open") ? close() : open();
    }
    if (e.key === "Escape") close();
  });
  backdrop.addEventListener("click", close);
  window.addEventListener("hashchange", close);
}

/**
 * Bindet Logout-Link an logout()-Funktion (lazy import).
 * @function initLogoutLink
 * @returns {void}
 */
export function initLogoutLink() {
  const logoutEl = document.getElementById("logoutLink");
  if (!logoutEl) return;
  logoutEl.addEventListener("click", async (e) => {
    e.preventDefault();
    try {
      const { logout } = await import("../auth.js");
      await logout();
    } catch (err) {
      console.error("Logout fehlgeschlagen", err);
    }
  });
}
