/**
 * Rendert die Summary-Kachelübersicht mit Begrüßung.
 * Wenn kein container übergeben wird, wird #layout-main genutzt.
 * @function renderSummaryView
 * @param {HTMLElement|string} [container] Optional DOM-Element oder CSS-Selector.
 * @returns {void}
 */
import { getActiveUser } from "../auth.js";

export function renderSummaryView(container) {
  let mount;
  if (container instanceof HTMLElement) {
    mount = container;
  } else if (typeof container === "string") {
    mount = document.querySelector(container);
  } else {
    mount = document.getElementById("layout-main");
  }
  if (!mount) return;

  // Dynamische Begrüßung + Nutzername
  let userName = "Guest";
  try {
    const user = getActiveUser && getActiveUser();
    if (user) {
      if (user.displayName && user.displayName.trim())
        userName = user.displayName.trim();
      else if (user.email) userName = user.email.split("@")[0];
    }
  } catch {}
  const hour = new Date().getHours();
  const greeting =
    hour < 12
      ? "Good morning,"
      : hour < 18
      ? "Good afternoon,"
      : "Good evening,";

  mount.innerHTML = `
