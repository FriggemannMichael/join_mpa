/**
 * Rendert die komplette Sidebar Navigation.
 * @function renderSidebar
 * @returns {string} HTML
 */
export function renderSidebar() {
  return /*html*/ `
    <nav aria-label="Primäre Navigation" class="app-sidebar">
      <img class="nav-logo" src="./img/icon/logo.svg" alt="Join Logo" />
      <div class="nav-menu" role="list">
        ${navLink({
          hash: "#summary",
          label: "Summary",
          icon: "summary_icon.svg",
          active: true,
          orderClass: "link-1",
        })}
        ${navLink({
          hash: "#board",
          label: "Board",
          icon: "board_icon.svg",
          orderClass: "link-2",
        })}
        ${navLink({
          hash: "#add-task",
          label: "Add Task",
          icon: "add_task_icon.svg",
          orderClass: "link-3",
        })}
        ${navLink({
          hash: "#contacts",
          label: "Contacts",
          icon: "contacts_icon.svg",
          orderClass: "link-4",
        })}
      </div>
      <div class="privacy-and-legal">
        <a href="#privacy"><p>Privacy Policy</p></a>
        <a href="#legal"><p>Legal Notice</p></a>
      </div>
    </nav>`;
}

/**
 * Erzeugt einen einzelnen Navigationslink.
 * @private
 * @param {Object} cfg Konfiguration
 * @param {string} cfg.hash Ziel-Hash
 * @param {string} cfg.label Anzeigename
 * @param {string} cfg.icon Icon-Dateiname
 * @param {boolean} [cfg.active=false] Aktiv-Status
 * @param {string} [cfg.orderClass=""] Zusätzliche Klassen
 * @returns {string} Link-HTML
 */
function navLink({ hash, label, icon, active = false, orderClass = "" }) {
  return /*html*/ `
    <a href="${hash}" class="nav-link ${orderClass} ${
    active ? "link-active" : ""
  }" data-nav>
      <div class="link-icon"><img src="./img/icon/${icon}" alt="${label} Icon" /></div>
      <div class="link-name"><p>${label}</p></div>
    </a>`;
}

/**
 * Markiert den aktiven Navigationslink anhand des Hash-Werts.
 * @function setActiveNav
 * @param {string} hash Hash wie '#summary'
 * @returns {void}
 */
export function setActiveNav(hash) {
  document.querySelectorAll("a.nav-link").forEach((a) => {
    if (a.getAttribute("href") === hash) a.classList.add("link-active");
    else a.classList.remove("link-active");
  });
}
